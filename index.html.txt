<h2>1. Feladat – Lista elemzés</h2>
<ol>
  <li>def lista_elemzes(lista1, lista2):</li>
  <li>    kozos = []</li>
  <li>    csak_elso = []</li>
  <li>    csak_masodik = []</li>
  <li></li>
  <li>    for elem in lista1:</li>
  <li>        if elem in lista2:</li>
  <li>            if elem not in kozos:</li>
  <li>                kozos.append(elem)</li>
  <li>        else:</li>
  <li>            if elem not in csak_elso:</li>
  <li>                csak_elso.append(elem)</li>
  <li></li>
  <li>    for elem in lista2:</li>
  <li>        if elem not in lista1:</li>
  <li>            if elem not in csak_masodik:</li>
  <li>                csak_masodik.append(elem)</li>
  <li></li>
  <li>    return kozos, csak_elso, csak_masodik</li>
  <li></li>
  <li>lista_a = ["mocskos", "egyetem", "Orbán", "Krupmli"]</li>
  <li>lista_b = ["Energia", "Lámpa", "Putyin", "Informatika"]</li>
  <li></li>
  <li>kozos_elemek, elso_lista_elemei, masodik_lista_elemei = lista_elemzes(lista_a, lista_b)</li>
  <li></li>
  <li>print("Közös elemek:", kozos_elemek)</li>
  <li>print("Csak az első listában:", elso_lista_elemei)</li>
  <li>print("Csak a második listában:", masodik_lista_elemei)</li>
</ol>

<h2>2. Feladat – Szó előfordulás</h2>
<ol>
  <li>def szo_elofordulas(szoveg):</li>
  <li>    szavak_szama = {}</li>
  <li></li>
  <li>    szavak = szoveg.split()</li>
  <li></li>
  <li>    for szo in szavak:</li>
  <li>        if szo in szavak_szama:</li>
  <li>            szavak_szama[szo] += 1</li>
  <li>        else:</li>
  <li>            szavak_szama[szo] = 1</li>
  <li></li>
  <li>    return szavak_szama</li>
  <li></li>
  <li>mondat = input("Írj be egy tetszőleges mondatot: ")</li>
  <li></li>
  <li>mondat_tisztitott = mondat.lower()</li>
  <li></li>
  <li>eredmeny = szo_elofordulas(mondat_tisztitott)</li>
  <li></li>
  <li>print("\nSzavak előfordulása:")</li>
  <li>print(eredmeny)</li>
</ol>
