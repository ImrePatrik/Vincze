1. Feladat – Lista elemzés

def lista_elemzes(lista1, lista2): kozos = [] csak_elso = [] csak_masodik = [] for elem in lista1: if elem in lista2: if elem not in kozos: kozos.append(elem) else: if elem not in csak_elso: csak_elso.append(elem) for elem in lista2: if elem not in lista1: if elem not in csak_masodik: csak_masodik.append(elem) return kozos, csak_elso, csak_masodik lista_a = ["mocskos", "egyetem", "Orbán", "Krupmli"] lista_b = ["Energia", "Lámpa", "Putyin", "Informatika"] kozos_elemek, elso_lista_elemei, masodik_lista_elemei = lista_elemzes(lista_a, lista_b) print("Közös elemek:", kozos_elemek) print("Csak az első listában:", elso_lista_elemei) print("Csak a második listában:", masodik_lista_elemei)

2. Feladat – Szó előfordulás

def szo_elofordulas(szoveg): szavak_szama = {} szavak = szoveg.split() for szo in szavak: if szo in szavak_szama: szavak_szama[szo] += 1 else: szavak_szama[szo] = 1 return szavak_szama mondat = input("Írj be egy tetszőleges mondatot: ") mondat_tisztitott = mondat.lower() eredmeny = szo_elofordulas(mondat_tisztitott) print("\nSzavak előfordulása:") print(eredmeny)
